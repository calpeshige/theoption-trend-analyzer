# インデックス方式（案C）詳細設計

## 基本コンセプト

データベースのインデックスと同じ考え方で、各属性ごとにインデックスを作成し、**積集合（AND検索）**で高速に絞り込む。

```
全データ（6000件）を最初から検索 ❌
  ↓
複数の条件で段階的に絞り込み ✅
```

## 積集合の計算方法

### アルゴリズム（効率的な実装）

```javascript
function findIntersection(arrays) {
  // 最も小さい配列から始める（効率化）
  arrays.sort((a, b) => a.length - b.length);

  if (arrays.length === 0) return [];
  if (arrays.length === 1) return arrays[0];

  // 最小の配列を基準にする
  let result = new Set(arrays[0]);

  // 他の配列と積集合を取る
  for (let i = 1; i < arrays.length; i++) {
    const currentSet = new Set(arrays[i]);
    result = new Set([...result].filter(id => currentSet.has(id)));
  }

  return Array.from(result);
}

// 使用例
candidates1 = [1, 15, 47, 99, 123, 156, ...];    // 500件
candidates2 = [1, 3, 9, 15, 27, 47, ...];        // 1500件
candidates3 = [1, 3, 11, 29, 47, 99, ...];       // 2000件
candidates4 = [1, 9, 27, 63, 99, 123, ...];      // 1500件

result = findIntersection([
  candidates1,
  candidates2,
  candidates3,
  candidates4
]);
// → [1, 47, 99, 123, ...] 80件
```

```javascript
function buildIndexes(allData) {
  console.log('📚 インデックス構築開始...');
  const startTime = Date.now();

  const indexes = {
    segment: {},
    rsi: {},
    macd: {},
    roc: {},
    stochastic: {},
    adx: {},
    trendStrength: {}
  };

  // 全データを走査（1回だけ）
  allData.forEach((data, index) => {
    // セグメントパターン
    const segmentKey = data.priceSegments60s.pattern;
    if (!indexes.segment[segmentKey]) {
      indexes.segment[segmentKey] = [];
    }
    indexes.segment[segmentKey].push(index);

    // RSI
    const rsiKey = classifyRSI(data.rsi);
    if (!indexes.rsi[rsiKey]) {
      indexes.rsi[rsiKey] = [];
    }
    indexes.rsi[rsiKey].push(index);

    // MACD
    const macdKey = classifyMACD(data.macdStrength);
    if (!indexes.macd[macdKey]) {
      indexes.macd[macdKey] = [];
    }
    indexes.macd[macdKey].push(index);

    // その他の指標も同様...
  });

  const buildTime = Date.now() - startTime;
  console.log(`✅ インデックス構築完了: ${buildTime}ms`);
  console.log(`📊 パターン数: セグメント=${Object.keys(indexes.segment).length}, RSI=3, MACD=3`);

  return indexes;
}

// 実行例
indexes = buildIndexes(allData); // 6000件
// → 構築時間: 約50-100ms（高速！）
```

## 実装例（擬似コード）

```javascript
class IndexedPatternMatcher {
  constructor() {
    this.data = [];
    this.indexes = {
      segment: {},
      rsi: {},
      macd: {},
      trendStrength: {}
    };
  }

  // インデックス構築（ボタン押下時）
  buildIndexes(data) {
    this.data = data;
    this.indexes = this._buildAllIndexes(data);
    console.log(`✅ インデックス構築完了: ${data.length}件`);
  }

  // 検索実行（2秒ごとの分析時）
  findSimilarPatterns(currentSituation, timeframe = 60) {
    // 1. 条件を抽出
    const conditions = {
      segment: currentSituation.priceSegments60s.pattern,
      rsi: this._classifyRSI(currentSituation.rsi),
      macd: this._classifyMACD(currentSituation.macdStrength),
      trendStrength: this._classifyTrend(currentSituation)
    };

    // 2. 各インデックスから候補を取得
    const candidateSets = [];

    if (this.indexes.segment[conditions.segment]) {
      candidateSets.push(this.indexes.segment[conditions.segment]);
    }

    if (this.indexes.rsi[conditions.rsi]) {
      candidateSets.push(this.indexes.rsi[conditions.rsi]);
    }

    if (this.indexes.macd[conditions.macd]) {
      candidateSets.push(this.indexes.macd[conditions.macd]);
    }

    if (this.indexes.trendStrength[conditions.trendStrength]) {
      candidateSets.push(this.indexes.trendStrength[conditions.trendStrength]);
    }

    // 3. 積集合を計算
    const matchingIndexes = this._intersection(candidateSets);
    console.log(`🔍 インデックス検索: ${candidateSets.length}条件 → ${matchingIndexes.length}件`);

    // マッチ数が少なすぎる場合は条件を緩める
    if (matchingIndexes.length < 30) {
      console.log('⚠️ マッチ数が少ないため条件を緩和');
      // セグメントとRSIだけで再検索
      const relaxedSets = [
        this.indexes.segment[conditions.segment],
        this.indexes.rsi[conditions.rsi]
      ].filter(Boolean);
      matchingIndexes = this._intersection(relaxedSets);
      console.log(`🔍 緩和後: ${matchingIndexes.length}件`);
    }

    // 4. 実データを取得
    const matchingData = matchingIndexes.map(idx => this.data[idx]);

    // 5. 詳細な類似度計算
    const withSimilarity = matchingData.map(data => ({
      data: data,
      similarity: this._calculateDetailedSimilarity(currentSituation, data, timeframe)
    }));

    // 6. 類似度でソート
    withSimilarity.sort((a, b) => b.similarity - a.similarity);

    // 7. 上位20件を返す
    return withSimilarity.slice(0, 20);
  }

  // 積集合計算
  _intersection(arrays) {
    if (arrays.length === 0) return [];
    if (arrays.length === 1) return arrays[0];

    // 最小の配列から始める
    arrays.sort((a, b) => a.length - b.length);
    let result = new Set(arrays[0]);

    for (let i = 1; i < arrays.length; i++) {
      const currentSet = new Set(arrays[i]);
      result = new Set([...result].filter(id => currentSet.has(id)));
    }

    return Array.from(result);
  }

  // 分類関数
  _classifyRSI(rsi) {
    if (rsi < 35) return "売られすぎ";
    if (rsi > 65) return "買われすぎ";
    return "中立";
  }

  _classifyMACD(macd) {
    if (macd < -0.015) return "下降";
    if (macd > 0.015) return "上昇";
    return "横ばい";
  }

  _classifyTrend(situation) {
    const adx = situation.adxValue;
    if (adx > 35) return "強い";
    if (adx < 20) return "弱い";
    return "中";
  }
}
```
